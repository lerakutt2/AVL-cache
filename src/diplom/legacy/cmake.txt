# Ищем quadmath.h
find_path(QUADMATH_INCLUDE_DIR
    NAMES quadmath.h
    PATHS
        /usr/include
        /usr/local/include
        /usr/lib/gcc/x86_64-linux-gnu/*/include
        /usr/lib/gcc/*/*/include
    PATH_SUFFIXES include
    NO_DEFAULT_PATH  # Не ищем в системных путях по умолчанию
)

# Ищем библиотеку quadmath с явным указанием путей
find_library(QUADMATH_LIBRARY
    NAMES quadmath
    PATHS
        /usr/lib
        /usr/lib/x86_64-linux-gnu
        /usr/local/lib
        /usr/lib/gcc/x86_64-linux-gnu/11
        /usr/lib/gcc/x86_64-linux-gnu/*  # любая версия
    PATH_SUFFIXES lib lib64
    NO_DEFAULT_PATH
)
